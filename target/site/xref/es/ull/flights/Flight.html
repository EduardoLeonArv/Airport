1   package es.ull.flights;
2   
3   import java.util.HashSet;
4   import java.util.Set;
5   import java.util.regex.Matcher;
6   import java.util.regex.Pattern;
7   import es.ull.passengers.Passenger;
8   
9   /**
10   * Clase que representa un vuelo.
11   */
12  public class Flight {
13  
14      private String flightNumber;    // Número de vuelo
15      private int seats;              // Número de asientos disponibles
16      private Set<Passenger> passengers = new HashSet<>();  // Conjunto de pasajeros
17  
18      private static String flightNumberRegex = "^[A-Z]{2}\\d{3,4}$"; // Expresión regular para validar el número de vuelo
19      private static Pattern pattern = Pattern.compile(flightNumberRegex);
20  
21      /**
22       * Constructor de la clase Flight.
23       *
24       * @param flightNumber Número del vuelo.
25       * @param seats Número de asientos disponibles.
26       * @throws RuntimeException Si el número del vuelo no es válido.
27       */
28      public Flight(String flightNumber, int seats) {
29          Matcher matcher = pattern.matcher(flightNumber);
30          if (!matcher.matches()) {
31              throw new RuntimeException("Invalid flight number");
32          }
33          this.flightNumber = flightNumber;
34          this.seats = seats;
35      }
36  
37      /**
38       * Obtiene el número del vuelo.
39       *
40       * @return Número del vuelo.
41       */
42      public String getFlightNumber() {
43          return flightNumber;
44      }
45  
46      /**
47       * Obtiene el número de pasajeros actualmente en el vuelo.
48       *
49       * @return Número de pasajeros.
50       */
51      public int getNumberOfPassengers() {
52          return passengers.size();
53      }
54  
55      /**
56       * Añade un pasajero al vuelo.
57       *
58       * @param passenger Pasajero a añadir.
59       * @return True si el pasajero se añadió correctamente.
60       * @throws RuntimeException Si no hay asientos disponibles.
61       */
62      public boolean addPassenger(Passenger passenger) {
63          if (getNumberOfPassengers() >= seats) {
64              throw new RuntimeException("Not enough seats for flight " + getFlightNumber());
65          }
66          passenger.setFlight(this);
67          return passengers.add(passenger);
68      }
69  
70      /**
71       * Elimina un pasajero del vuelo.
72       *
73       * @param passenger Pasajero a eliminar.
74       * @return True si el pasajero se eliminó correctamente.
75       */
76      public boolean removePassenger(Passenger passenger) {
77          passenger.setFlight(null);
78          return passengers.remove(passenger);
79      }
80  }
