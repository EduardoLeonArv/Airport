1   package es.ull.passengers;
2   
3   import java.util.Arrays;
4   import java.util.Locale;
5   import es.ull.flights.Flight;
6   
7   /**
8    * Clase que representa a un pasajero.
9    */
10  public class Passenger {
11  
12      private String identifier;   // Identificador único del pasajero
13      private String name;         // Nombre del pasajero
14      private String countryCode;  // Código del país del pasajero (en formato ISO)
15      private Flight flight;       // Vuelo asociado al pasajero
16  
17      /**
18       * Constructor de la clase Passenger.
19       *
20       * @param identifier Identificador único del pasajero.
21       * @param name Nombre del pasajero.
22       * @param countryCode Código del país del pasajero (en formato ISO).
23       * @throws RuntimeException Si el código de país no es válido.
24       */
25      public Passenger(String identifier, String name, String countryCode) {
26          if (!Arrays.asList(Locale.getISOCountries()).contains(countryCode)) {
27              throw new RuntimeException("Invalid country code");
28          }
29  
30          this.identifier = identifier;
31          this.name = name;
32          this.countryCode = countryCode;
33      }
34  
35      /**
36       * Obtiene el identificador del pasajero.
37       *
38       * @return Identificador del pasajero.
39       */
40      public String getIdentifier() {
41          return identifier;
42      }
43  
44      /**
45       * Obtiene el nombre del pasajero.
46       *
47       * @return Nombre del pasajero.
48       */
49      public String getName() {
50          return name;
51      }
52  
53      /**
54       * Obtiene el código de país del pasajero.
55       *
56       * @return Código del país del pasajero.
57       */
58      public String getCountryCode() {
59          return countryCode;
60      }
61  
62      /**
63       * Obtiene el vuelo al que está asociado el pasajero.
64       *
65       * @return Vuelo asociado al pasajero.
66       */
67      public Flight getFlight() {
68          return flight;
69      }
70  
71      /**
72       * Asigna un pasajero a un vuelo.
73       *
74       * @param flight Vuelo al que el pasajero se unirá.
75       * @throws RuntimeException Si no se puede eliminar o añadir el pasajero del/ al vuelo.
76       */
77      public void joinFlight(Flight flight) {
78          Flight previousFlight = this.flight;
79          if (null != previousFlight) {
80              if (!previousFlight.removePassenger(this)) {
81                  throw new RuntimeException("Cannot remove passenger");
82              }
83          }
84          setFlight(flight);
85          if (null != flight) {
86              if (!flight.addPassenger(this)) {
87                  throw new RuntimeException("Cannot add passenger");
88              }
89          }
90      }
91  
92      /**
93       * Asigna un vuelo al pasajero.
94       *
95       * @param flight Vuelo a asignar al pasajero.
96       */
97      public void setFlight(Flight flight) {
98          this.flight = flight;
99      }
100 
101     /**
102      * Representación en cadena de la información del pasajero.
103      *
104      * @return Información del pasajero en formato de cadena.
105      */
106     @Override
107     public String toString() {
108         return "Passenger " + getName() + " with identifier: " + getIdentifier() + " from " + getCountryCode();
109     }
110 }
